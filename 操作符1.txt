#include<stdio.h>

int main()
{
	int a = 10;
	char c = 'cx';
	char* p = &c;  
	int arr[10] = { 0 };

	printf("%d\n", sizeof(a));//4
	printf("%d\n", sizeof(int));

	printf("%d\n", sizeof(c));//1
	printf("%d\n", sizeof(char));

	printf("%d\n", sizeof(p));//4
	printf("%d\n", sizeof(char*));

	printf("%d\n", sizeof(arr));//40
	printf("%d\n", sizeof(int [10]));//数组类型
	return 0;
}
